    code fragment is:

    max = z;
    if ( x + y > z )

    c) max = (x + y > z) : x + y ? z;

    a)  if(c >= '0')

            if(c <= '9');
                printf("This is a number.\n");

    c)  if(c >= '0')
            if(c <= '9')
                printf("This is a number.\n");
